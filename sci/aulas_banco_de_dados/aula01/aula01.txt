a PK da outra tabela, deixa o campo FK nulo):*/

/*Consulta:*/
SELECT U.BDCODUSUARIO CÓDIGO, 
       UPPER(U.BDNOMEUSUARIO) NOME,
       U.BDDATANASCIMENTO DATA_NASCIMENTO, 
       15 BDVALOR
FROM TUSUARIO_TREINAMENTO U
WHERE U.BDDATANASCIMENTO >= '03/25/1999';

ou
WHERE U.BDDATANASCIMENTO BETWEEN '03/25/1999' AND '12/25/2007';

***Collate WIN_PTBR facilita a consulta por nomes


/*Consulta com JOIN:*/
SELECT U.BDCODUSUARIO CÓDIGO, 
       UPPER(U.BDNOMEUSUARIO) NOME,
       UF.BDSIGLAUF SIGLA
FROM TUSUARIO_TREINAMENTO U
JOIN TUF UF ON (U.BDCODUF = UF.BDCODUF)


/*Consulta com GROUP BY:*/
SELECT MAX(U.BDCODUSUARIO)
       U.BDNOMEUSUARIO,
       U.BDCODUF,
       U.BDDATANASCIMENTO,
FROM TUSUARIO_TREINAMENTO U
LEFT OUTER JOIN TUF UF ON (U.BDCODUF = UF.BDCODUF)
GROUP BY U.BDNOMEUSUARIO, U.BDCODUF, U.BDDATANASCIMENTO
***GROUP BY agrupa os usuários com os mesmos nomes,
   códigos de uf e datas de nascimento


/*Consulta com ORDER BY:*/
SELECT MAX(U.BDCODUSUARIO)
       U.BDNOMEUSUARIO,
       U.BDCODUF,
       U.BDDATANASCIMENTO,
FROM TUSUARIO_TREINAMENTO U
LEFT OUTER JOIN TUF UF ON (U.BDCODUF = UF.BDCODUF)
ORDER BY U.BDCODUSUARIO DESC
