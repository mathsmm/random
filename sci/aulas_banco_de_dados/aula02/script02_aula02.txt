/*Trazer os dados dos clientes cadastrados e a sigla do estado*/
SELECT C.BDCODCLIENTE CODIGO,
       C.BDNOME NOME,
       C.BDCNPJCPF CNPJCPF,
       C.BDENDERECO ENDERECO,
       C.BDTELEFONE TELEFONE,
       C.BDEMAIL EMAIL,
       UF.BDSIGLAUF SIGLA
FROM TCLIENTE C
JOIN TUF UF ON (C.BDCODUF = UF.BDCODUF);

/*Trazer os clientes do estado de SC*/
SELECT C.BDCODCLIENTE CODIGO,
       C.BDNOME NOME,
       C.BDCNPJCPF CNPJCPF,
       C.BDENDERECO ENDERECO,
       C.BDTELEFONE TELEFONE,
       C.BDEMAIL EMAIL,
       UF.BDSIGLAUF SIGLA
FROM TCLIENTE C
JOIN TUF UF ON (C.BDCODUF = UF.BDCODUF)
WHERE UF.BDSIGLAUF = 'SC';


/*Trazer somente os clientes que são Pessoas Físicas - CPF*/
SELECT C.BDCODCLIENTE CODIGO,
       C.BDNOME NOME,
       C.BDCNPJCPF CNPJCPF,
       C.BDENDERECO ENDERECO,
       C.BDTELEFONE TELEFONE,
       C.BDEMAIL EMAIL,
       UF.BDSIGLAUF SIGLA
FROM TCLIENTE C
JOIN TUF UF ON (C.BDCODUF = UF.BDCODUF)
WHERE CHAR_LENGTH(C.BDCNPJCPF) = 14;


/*Trazer somente os clientes que são Pessoas Jurídicas - CNPJ*/

       C.BDCNPJCPF CNPJCPF,
       C.BDENDERECO ENDERECO,
       C.BDTELEFONE TELEFONE,
       C.BDEMAIL EMAIL,
       UF.BDSIGLAUF SIGLA
FROM TCLIENTE C
